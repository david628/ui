var dialog = new DBapp.ui.Dialog("id-dlg", {
  title: '提示',
  msg: 'Hello'
});
dialog.setWidth(500);
dialog.clearButton();
dialog.addButton([{
  text : "确定",//按钮名称
  cls : 'submit',//按钮样式
  handle : function(event) {//点击按钮触发方法
    alert("确定");
  }
}, {
  text : "取消",//按钮名称
  cls : 'cancel',//按钮样式
  handle : function(event) {//点击按钮触发方法
    alert("取消");
    dialog.hide();
  }
}]);
dialog.show();
属性
openAnimationDuration (类型: Number)

设置弹出对话框时，动画时间.

closeAnimationDuration (类型: Number)

设置关闭对话框时，动画时间.

openAnimationEffect (类型: String): default-in, rollIn, fadeInLeft

设置弹出对话框时，动画特效.

openAnimationEffect (类型: String): default-out, rollOut, fadeOutRight

设置关闭对话框时，动画特效.

autoClose (类型: Boolean): 默认为false

设置弹出对话框后，是否自动关闭.

buttonAlign (类型: String): left, center, right 默认为false

设置按钮对齐方式(左，中，右).

方法
setButtonAlign (String pos): void (left, center, right)

设置按钮对齐方式(左，中，右).

setWidth (Number width): void

设置宽.

setHeight (Number height): void

设置高.

setSize (Number width, Number height): void

设置宽和高.

setTitle (String title): void

设置标题.

setContent (String msg): void

设置内容.

clearButton (): void

清空所有按扭.

addButton (Array btns): void

添加按扭.
