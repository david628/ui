id-test 


var tip = new DBapp.ui.Tip({
  position : "left",//提示方向(left, top, right, bottom)
  msg : "2015",//提示内容
  distance : 10
});
jQuery("#id-test").hover(function() {
  tip.show(this, new Date(), "right");//第一参数：参照物Dom，第二参数：显示内容，第三参数：显示的位置
}, function() {
  tip.hide();//隐藏
});
